[1561. 你可以获得的最大硬币数目](https://leetcode.cn/problems/maximum-number-of-coins-you-can-get/description/)
```
class Solution {
public:
    int maxCoins(vector<int>& piles) {
        int n=piles.size();
        int m=n/3; m*=2;
        sort(piles.begin(),piles.end(),greater<int>());
        int ans=0;
        for(int i=1;i<m;i+=2){
            ans+=piles[i];
        }
        return ans;
    }
};
```
