[2266. 统计打字方案数](https://leetcode.cn/problems/count-number-of-texts/description/)
```cpp
class Solution {
public:
    int countTexts(string pressedKeys) {
        long long ans=1;
        int n=pressedKeys.size();
        long long f[10][n+10];
        memset(f,0ll,sizeof f);
        const int N=1e9+7;
        for(int i=2;i<=9;i++){
            if(i!=7&&i!=9){
                f[i][0]=1;
                f[i][1]=1;
                f[i][2]=2;
                f[i][3]=4;
                for(int j=4;j<=n;j++){
                    f[i][j]=f[i][j-1]+f[i][j-2]+f[i][j-3];
                    f[i][j]%=N;
                }
            }else{
                f[i][0]=1;
                f[i][1]=1;
                f[i][2]=2;
                f[i][3]=4;
                f[i][4]=8;
                for(int j=4;j<=n;j++){
                    f[i][j]=f[i][j-1]+f[i][j-2]+f[i][j-3]+f[i][j-4];
                    f[i][j]%=N;
                }
            }
            
        }

        int t=1;
        for(int i=1;i<n;i++){
            long long res=1;
            if(pressedKeys[i]==pressedKeys[i-1]){
                t++;
            }else{
                ans=(ans*f[pressedKeys[i-1]-'0'][t])%N;
                t=1;
            }
        }

        ans=(ans*f[pressedKeys[n-1]-'0'][t])%N;
        return ans;
    }
};
```
